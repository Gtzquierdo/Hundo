The Code for Hundo


                        // Get the values from the Page
// starts or controller function
function getValues() {
    //get values from the page
    let startValue = document.getElementById("startValue").value;
    let endValue = document.getElementById("endValue").value;


    //We need to validate our inputs
    //parse into Integers
    startValue = parseInt(startValue);
    endValue = parseInt(endValue);

    if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
        //we call generateNumbers
        let numbers = generateNumbers(startValue, endValue);   
        //we call displayNumbers
        displayNumbers(numbers);
    } else {
        //display error message
        alert("Please enter a valid integer");
    }


}   

// Generate numbers from startValue to the endValue
// logic function(s)
function generateNumbers(sValue, eValue) {
    let numbers = [];

    //we want to get all numbers from start to end
    for (let i = sValue; i <= eValue; i++) {

        //this will execute in a loop until index= eValue
        numbers.push(i);
    }

    return numbers;
}

// Display the numbers and mark even numbers bold
// display or view functions
function displayNumbers(numbers){

    let templateRows = "";
    for (let index = 0; index < numbers.length; index++) {

        let className = "even";

        let number = numbers[index];

        if(number % 2 === 0){
            className = "even";
        } else{
            className = "odd";
        }
        // This does render correctly with Prism, see the source code
        templateRows += `<tr><td class="${className}" >${number}</td></tr>

    }

    document.getElementById("results").innerHTML = templateRows;
}

The Code is structured in functions.

function getValues

A function where we get values from the page, validate the inputs, and parse into integers. Then if number is true we call generateNumbers and displayNumbers. Else we alert the user that the they need to enter a valid integer.

function generateNumbers

This function is to generate numbers from the startValue to the endValue. We have a for loop to get all the numbers from start to end. We will continue the loop until index or i in this case = eValue (endValue)

function displayNumbers

Display the number and BOLD the numbers that are even. The numbers displayed are placed in rows below on the app page.
